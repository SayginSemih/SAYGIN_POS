create database cafeteria;

use cafeteria;

CREATE TABLE log (
    id INT(11) NOT NULL AUTO_INCREMENT,
    urun VARCHAR(150) CHARACTER SET utf8mb4 COLLATE utf8mb4_turkish_ci NOT NULL,
    fiyat INT(11) NOT NULL,
    PRIMARY KEY (id)
);

CREATE TABLE orders(
    id INT(11) NOT NULL AUTO_INCREMENT,
    urun VARCHAR(150) CHARACTER SET utf8mb4 COLLATE utf8mb4_turkish_ci NOT NULL,
    fiyat INT(11) NOT NULL,
    category INT(11) NOT NULL,
    masa INT(11) NOT NULL,
    ip varchar(25),
    musterinotu varchar(150),
    durum varchar(25),
    PRIMARY KEY (id)
);

CREATE TABLE product(
    id INT(11) NOT NULL AUTO_INCREMENT,
    urun VARCHAR(150) CHARACTER SET utf8mb4 COLLATE utf8mb4_turkish_ci NOT NULL,
    fiyat INT(11) NOT NULL,
    category INT(11) NOT NULL,
    PRIMARY KEY (id)
);

CREATE TABLE users(
    id INT(11) NOT NULL AUTO_INCREMENT,
    username VARCHAR(150) CHARACTER SET utf8mb4 COLLATE utf8mb4_turkish_ci NOT NULL,
    passwd INT(11) NOT NULL,
    PRIMARY KEY (id)
);
